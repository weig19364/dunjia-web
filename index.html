Astronomy def get_solar_lon(dt): # Simplified VSOP87 y, m, d = dt.year, dt.month, dt.day h = dt.hour + dt.minute/60.0 + dt.second/3600.0 if m <= 2: y -= 1; m += 12 a = int(y / 100); b = 2 - a + int(a / 4) jd = int(365.25 * (y + 4716)) + int(30.6001 * (m + 1)) + d + h/24.0 + b - 1524.5 tau = (jd - 2451545.0) / 365250.0 L0 = 175347046.0 + 3341656.0 * math.cos(4.6692568 + 6283.07585 * tau) L = (L0 * 1e-8 + math.radians(280.46646 + 36000.76983 * tau)) return (math.degrees(L) + 0.0) % 360.0 def get_true_solar_time(dt, lon): doy = dt.timetuple().tm_yday b = (2 * math.pi / 365.24) * (doy - 81) eot = 9.87 * math.sin(2*b) - 7.53 * math.cos(b) - 1.5 * math.sin(b) tst = dt + datetime.timedelta(minutes=eot + (lon-120)*4) return tst, eot # 3.
